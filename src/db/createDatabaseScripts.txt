CREATE TABLE [User] (
	ID UNIQUEIDENTIFIER PRIMARY KEY NOT NULL,
	SpotifyAccountID NVARCHAR(100) UNIQUE,
	Email NVARCHAR(50) NOT NULL UNIQUE, 
	Username NVARCHAR(50) NOT NULL UNIQUE, 
	[Password] NVARCHAR(150) NOT NULL
)

CREATE TABLE [Room] (
	ID UNIQUEIDENTIFIER PRIMARY KEY NOT NULL,
	[Name] NVARCHAR(50) NOT NULL,
	OwnerID NVARCHAR(50) NOT NULL,
	LastPlayed NVARCHAR(50),
	PlaybackMS NVARCHAR(30),
	IsPlaying BIT NOT NULL
)

CREATE TABLE User_Room (
	ID UNIQUEIDENTIFIER PRIMARY KEY NOT NULL,
	Ref_User_ID UNIQUEIDENTIFIER NOT NULL,
	Ref_Room_ID UNIQUEIDENTIFIER NOT NULL
)
